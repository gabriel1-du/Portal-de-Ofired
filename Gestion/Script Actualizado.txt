CREATE DATABASE IF NOT EXISTS gasfiteria_db;
USE gasfiteria_db;

-- =========================================================================
-- CONFIGURACIONES BASE / TABLAS MAESTRAS
-- =========================================================================

CREATE TABLE region (
    id_region INT AUTO_INCREMENT PRIMARY KEY,
    nombre_region VARCHAR(100) NOT NULL
);

CREATE TABLE oficio (
    id_oficio INT AUTO_INCREMENT PRIMARY KEY,
    nombre_oficio VARCHAR(100) NOT NULL
);

CREATE TABLE medio_de_pago (
    id_medio_pago INT AUTO_INCREMENT PRIMARY KEY,
    nombre_medio_pago VARCHAR(100) NOT NULL
);

CREATE TABLE tipo_de_trabajo (
    id_tipo_trabajo INT AUTO_INCREMENT PRIMARY KEY,
    nombre_tipo_trabajo VARCHAR(100) NOT NULL
);

CREATE TABLE tipo_usuario (
    id_tipo_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nombre_rol VARCHAR(50) NOT NULL
);

CREATE TABLE sexo_usuario (
    id_sexo INT AUTO_INCREMENT PRIMARY KEY,
    nombre_sexo VARCHAR(50) NOT NULL
);

CREATE TABLE tipo_denuncia (
    id_tipo_denuncia INT AUTO_INCREMENT PRIMARY KEY,
    nombre_tipo_denuncia VARCHAR(100) NOT NULL
);

CREATE TABLE tipo_contenido_denunciado (
    id_tipo_contenido INT AUTO_INCREMENT PRIMARY KEY,
    nombre_contenido VARCHAR(100) NOT NULL
);

CREATE TABLE comuna (
    id_comuna INT AUTO_INCREMENT PRIMARY KEY,
    id_region INT NOT NULL,
    nombre_comuna VARCHAR(100) NOT NULL,
    CONSTRAINT fk_comuna_region FOREIGN KEY (id_region) REFERENCES region(id_region)
);

-- =========================================================================
-- ENTIDADES PRINCIPALES (USUARIOS Y PERFILES)
-- =========================================================================

CREATE TABLE usuario (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    p_nombre VARCHAR(50) NOT NULL,
    s_nombre VARCHAR(50),
    p_apellido VARCHAR(50) NOT NULL,
    s_apellido VARCHAR(50),
    id_sexo_usu INT NOT NULL,
    correo_elec VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    id_tipo_usu INT NOT NULL,
    rut VARCHAR(12) NULL UNIQUE,
    rut_dv VARCHAR(1) NULL,
    numero_telef VARCHAR(20),
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    foto VARCHAR(255),
    valoracion DECIMAL(3,2) NOT NULL DEFAULT 0.00,
    id_region_usu INT,
    id_comuna_usu INT,
    habilitador_administrador BOOLEAN DEFAULT FALSE,
    id_oficio INT,
    CONSTRAINT fk_usu_sexo FOREIGN KEY (id_sexo_usu) REFERENCES sexo_usuario(id_sexo),
    CONSTRAINT fk_usu_tipo FOREIGN KEY (id_tipo_usu) REFERENCES tipo_usuario(id_tipo_usuario),
    CONSTRAINT fk_usu_region FOREIGN KEY (id_region_usu) REFERENCES region(id_region),
    CONSTRAINT fk_usu_comuna FOREIGN KEY (id_comuna_usu) REFERENCES comuna(id_comuna),
    CONSTRAINT fk_usu_oficio FOREIGN KEY (id_oficio) REFERENCES oficio(id_oficio)
);

CREATE TABLE perfil_usuario (
    id_perfil_usuario INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL UNIQUE, 
    nombre_apodo VARCHAR(100),
    p_nombre VARCHAR(50),
    s_nombre VARCHAR(50),
    p_apellido VARCHAR(50),
    s_apellido VARCHAR(50),
    numero_telefono VARCHAR(20),
    foto_perfil VARCHAR(255) NULL,
    fotografia_banner VARCHAR(255) NULL,
    descripcion VARCHAR(500) NULL,
    id_region INT,
    id_comuna INT,
    id_oficio INT,
    calificacion_p_usuario DECIMAL(3,2) DEFAULT 0.00,
    sexo_usu_per INT,
    CONSTRAINT fk_perfil_usuario FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_perfil_region FOREIGN KEY (id_region) REFERENCES region(id_region),
    CONSTRAINT fk_perfil_comuna FOREIGN KEY (id_comuna) REFERENCES comuna(id_comuna),
    CONSTRAINT fk_perfil_oficio FOREIGN KEY (id_oficio) REFERENCES oficio(id_oficio),
    CONSTRAINT fk_perfil_sexo FOREIGN KEY (sexo_usu_per) REFERENCES sexo_usuario(id_sexo)
);

-- =========================================================================
-- FASE 4: MARKETPLACE (PUBLICACIONES Y LAS 2 TABLAS DE COMENTARIOS)
-- =========================================================================

CREATE TABLE publicacion (
    id_publicacion INT AUTO_INCREMENT PRIMARY KEY,
    id_autor INT NOT NULL,
    titulo_publicacion VARCHAR(150) NOT NULL,
    id_region INT NOT NULL,
    id_comuna INT NOT NULL,
    ubicacion_publicacion VARCHAR(255) DEFAULT 'indefinido',
    descripcion_publicacion TEXT NOT NULL,
    fecha_creacion DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    fecha_publicacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    cantidad_likes INT DEFAULT 0,
    CONSTRAINT fk_publi_autor FOREIGN KEY (id_autor) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_publi_region FOREIGN KEY (id_region) REFERENCES region(id_region),
    CONSTRAINT fk_publi_comuna foreign key (id_comuna) REFERENCES comuna(id_comuna)
);


CREATE TABLE fotos_publi (
    id_foto_publi INT AUTO_INCREMENT PRIMARY KEY,
    id_publicacion INT NOT NULL,
    url_foto VARCHAR(255) NOT NULL,
    CONSTRAINT fk_foto_publi FOREIGN KEY (id_publicacion) REFERENCES publicacion(id_publicacion) ON DELETE CASCADE
);

-- 🆕 1. TABLA COMENTARIO (Sincronizada con tu Workbench)
CREATE TABLE comentario (
    id_comentario INT AUTO_INCREMENT PRIMARY KEY,
    id_publicacion INT NOT NULL,
    id_usuario INT NOT NULL,
    contenido TEXT NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_comentario_publi FOREIGN KEY (id_publicacion) REFERENCES publicacion(id_publicacion) ON DELETE CASCADE,
    CONSTRAINT fk_comentario_usu FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario) ON DELETE CASCADE
);

-- 🆕 2. TABLA RESPUESTA_COMENTARIO (Sincronizada con tu Workbench)
CREATE TABLE respuesta_comentario (
    id_respuesta INT AUTO_INCREMENT PRIMARY KEY,
    id_comentario INT NOT NULL UNIQUE,
    id_usuario_respuesta INT NOT NULL,
    contenido_respuesta TEXT NOT NULL,
    fecha_respuesta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_resp_coment_origen FOREIGN KEY (id_comentario) REFERENCES comentario(id_comentario) ON DELETE CASCADE,
    CONSTRAINT fk_resp_coment_usu FOREIGN KEY (id_usuario_respuesta) REFERENCES usuario(id_usuario) ON DELETE CASCADE
);

CREATE TABLE resenia (
    id_resenia INT AUTO_INCREMENT PRIMARY KEY,
    id_autor INT NOT NULL,
    id_usuario_reseniado INT NOT NULL,
    calificacion DECIMAL(3,2) NOT NULL,
    texto_resenia TEXT NOT NULL,
    fecha_creacion DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_resenia_autor FOREIGN KEY (id_autor) REFERENCES usuario(id_usuario),
    CONSTRAINT fk_resenia_destinatario FOREIGN KEY (id_usuario_reseniado) REFERENCES usuario(id_usuario) ON DELETE CASCADE
);

CREATE TABLE respuesta_resenia (
    id_respuesta_resenia INT AUTO_INCREMENT PRIMARY KEY,
    id_resenia INT NOT NULL,
    id_autor_res INT NOT NULL,
    texto_respuesta_resenia TEXT NOT NULL,
    fecha_creacion DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_resp_resenia FOREIGN KEY (id_resenia) REFERENCES resenia(id_resenia) ON DELETE CASCADE,
    CONSTRAINT fk_resp_autor FOREIGN KEY (id_autor_res) REFERENCES usuario(id_usuario)
);

-- =========================================================================
-- COMUNICACIONES, DENUNCIAS Y BLOQUEOS
-- =========================================================================

CREATE TABLE chat (
    id_chat INT AUTO_INCREMENT PRIMARY KEY,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE participantes_chat (
    id_participantes_chat INT AUTO_INCREMENT PRIMARY KEY,
    id_chat INT NOT NULL,
    id_usuario_participante INT NOT NULL,
    CONSTRAINT fk_participante_chat FOREIGN KEY (id_chat) REFERENCES chat(id_chat) ON DELETE CASCADE,
    CONSTRAINT fk_participante_usuario FOREIGN KEY (id_usuario_participante) REFERENCES usuario(id_usuario) ON DELETE CASCADE
);

CREATE TABLE mensajes_chat (
    id_mensaje_chat INT AUTO_INCREMENT PRIMARY KEY,
    id_chat INT NOT NULL,
    id_autor_mensaje INT NOT NULL,
    mensaje_texto TEXT NOT NULL,
    fecha_hora_envio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_mensaje_chat FOREIGN KEY (id_chat) REFERENCES chat(id_chat) ON DELETE CASCADE,
    CONSTRAINT fk_mensaje_autor FOREIGN KEY (id_autor_mensaje) REFERENCES usuario(id_usuario)
);

CREATE TABLE denuncias (
    id_denuncia INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario_denunciante INT NOT NULL,
    id_tipo_denuncia INT NOT NULL,
    id_tipo_contenido INT NOT NULL,
    descripcion_denuncia TEXT NOT NULL,
    fecha_denuncia TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    id_usuario_denunciado INT NULL,
    id_publicacion_denunciada INT NULL,
    id_resenia_denunciada INT NULL,      
    id_respuesta_denunciada INT NULL,    
    
    CONSTRAINT fk_denuncia_autor FOREIGN KEY (id_usuario_denunciante) REFERENCES usuario(id_usuario),
    CONSTRAINT fk_denuncia_tipo FOREIGN KEY (id_tipo_denuncia) REFERENCES tipo_denuncia(id_tipo_denuncia),
    CONSTRAINT fk_denuncia_contenido FOREIGN KEY (id_tipo_contenido) REFERENCES tipo_contenido_denunciado(id_tipo_contenido),
    
    CONSTRAINT fk_denuncia_usu_target FOREIGN KEY (id_usuario_denunciado) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_denuncia_publi_target FOREIGN KEY (id_publicacion_denunciada) REFERENCES publicacion(id_publicacion) ON DELETE CASCADE,
    CONSTRAINT fk_denuncia_resenia_target FOREIGN KEY (id_resenia_denunciada) REFERENCES resenia(id_resenia) ON DELETE CASCADE,
    CONSTRAINT fk_denuncia_resp_target FOREIGN KEY (id_respuesta_denunciada) REFERENCES respuesta_resenia(id_respuesta_resenia) ON DELETE CASCADE
);

CREATE TABLE usuarios_bloqueados (
    id_bloqueo INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario_que_bloquea INT NOT NULL,
    id_usuario_bloqueado INT NOT NULL,
    habilitador BOOLEAN DEFAULT TRUE,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_bloqueo_autor FOREIGN KEY (id_usuario_que_bloquea) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_bloqueo_destino FOREIGN KEY (id_usuario_bloqueado) REFERENCES usuario(id_usuario) ON DELETE CASCADE
);

-- =========================================================================
-- FASE 5: HISTORIAL DE TRATOS / TRANSACCIONES
-- =========================================================================

CREATE TABLE confirmacion_transaccion (
    id_transaccion INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario_oferente INT NOT NULL,
    id_usuario_cliente INT NOT NULL,
    monto_servicio DECIMAL(10,2) NOT NULL,
    id_medio_pago INT NOT NULL,
    id_tipo_trabajo INT NOT NULL,
    aceptado BOOLEAN DEFAULT NULL, 
    observaciones_trato TEXT, 
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
    CONSTRAINT fk_trans_oferente FOREIGN KEY (id_usuario_oferente) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_trans_cliente FOREIGN KEY (id_usuario_cliente) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    CONSTRAINT fk_trans_medio_pago FOREIGN KEY (id_medio_pago) REFERENCES medio_de_pago(id_medio_pago),
    CONSTRAINT fk_trans_tipo_trabajo FOREIGN KEY (id_tipo_trabajo) REFERENCES tipo_de_trabajo(id_tipo_trabajo)
);
